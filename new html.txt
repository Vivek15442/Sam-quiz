    function validateName() {
        const nameEl = document.getElementById('su-name');
        const hint = document.getElementById('hint-name');
        if (!nameEl || !hint) return false;
        const val = nameEl.value.trim();
        if (!val) { hint.textContent = ''; hint.className = 'input-hint'; return false; }

        hint.innerHTML = '<i class="fas fa-circle-check"></i> Looks good'; hint.className = 'input-hint success'; return true;
    }
 
    function validateEmail(fieldId, hintId) {
        const fieldEl = document.getElementById(fieldId);
        const hint = document.getElementById(hintId);
        if (!fieldEl || !hint) return false;
        const val = fieldEl.value.trim();
        if (!val) { hint.textContent = ''; hint.className = 'input-hint'; return false; }
        const ok = /^[^\s@]+@[^\s@]+\.[^\s@]+$/.test(val);
        if (!ok) { hint.innerHTML = '<i class="fas fa-circle-xmark"></i> Enter a valid email'; hint.className = 'input-hint error'; return false; }
        hint.innerHTML = '<i class="fas fa-circle-check"></i> Valid email'; hint.className = 'input-hint success'; return true;
    }
 
    function checkPwStrength() {
        const pwEl = document.getElementById('su-pw');
        if (!pwEl) return 0;
        const pw = pwEl.value;
        const bars = ['bar1','bar2','bar3','bar4'].map(id => document.getElementById(id)).filter(Boolean);
        const label = document.getElementById('pw-label');
 
        // Update requirement checks
        const setReq = (id, met) => {
            const el = document.getElementById(id);
            if (el) el.classList.toggle('met', met);
        };
        setReq('req-len',     pw.length >= 6);
        setReq('req-upper',   /[A-Z]/.test(pw));
        setReq('req-num',     /[0-9]/.test(pw));
        setReq('req-special', /[^A-Za-z0-9]/.test(pw));
 
        bars.forEach(b => b.className = 'pw-bar');
        if (!pw) { if (label) label.textContent = 'Enter a password'; return 0; }
        let score = 0;
        if (pw.length >= 6)  score++;
        if (pw.length >= 10) score++;
        if (/[A-Z]/.test(pw) || /[0-9]/.test(pw)) score++;
        if (/[^A-Za-z0-9]/.test(pw)) score++;
        const cls = score <= 2 ? 'weak' : score === 3 ? 'fair' : 'strong';
        const labels = { weak: '⚠ Weak', fair: '~ Fair', strong: '✓ Strong' };
        for (let i = 0; i < score; i++) { if (bars[i]) bars[i].classList.add(cls); }
        if (label) label.textContent = labels[cls] || '';
        return score;
    }

    function loadQuestion() {
        if (!state.quiz.active) return;
        state.quiz.isAnswering = false;
 
        const prog = document.getElementById('quiz-progress');
        if (prog) {
            prog.innerHTML = '';
            for (let i = 0; i < 5; i++) {
                const d = document.createElement('div');
                d.className = 'prog-dot' + (i <= state.quiz.currentIdx ? ' active' : '');
                prog.appendChild(d);
            }
        }
 
        const q = BANK[Math.floor(Math.random() * BANK.length)];
        const qEl = document.getElementById('quiz-q');
        if (qEl) qEl.innerText = q.q;
 
        const box = document.getElementById('quiz-options');
        if (!box) return;
        box.innerHTML = '';
        q.o.forEach((opt, i) => {
            const btn = document.createElement('button');
            btn.className = 'opt-btn';
            btn.innerText = opt;
            btn.onclick = () => {
                if (state.quiz.isAnswering) return;
                state.quiz.isAnswering = true;
                clearInterval(state.quiz.interval);
                if (i === q.c) {
                    btn.classList.add('correct');
                    setTimeout(() => {
                        if (state.quiz.currentIdx < 4) {
                            state.quiz.currentIdx++;
                            loadQuestion();
                        } else finishGame(true);
                    }, 700);
                } else {
                    btn.classList.add('wrong');
                    setTimeout(() => finishGame(false, 'Incorrect answer. Keep practicing!'), 700);
                }
            };
            box.appendChild(btn);
        });
 
        resetTimer();
    }
 
    function resetTimer() {
        if (state.quiz.interval) clearInterval(state.quiz.interval);
        state.quiz.timer = 60;
        const timerEl = document.getElementById('quiz-timer');
        if (timerEl) timerEl.innerText = '60';
        state.quiz.interval = setInterval(() => {
            state.quiz.timer--;
            const t = document.getElementById('quiz-timer');
            if (t) t.innerText = state.quiz.timer;
            if (state.quiz.timer <= 0) finishGame(false, 'Time ran out! Try again.');
        }, 1000);
    }
 
    function finishGame(win, reason = '') {
        clearInterval(state.quiz.interval);
        state.quiz.active = false;
        switchView('result');
 
        const icon = document.getElementById('res-icon');
        const reward = document.getElementById('res-reward');

            state.wallet += 0.5;
            state.totalEarned += 0.5;

            icon.innerHTML = '<i class="fas fa-check"></i>';

            document.getElementById('res-msg').innerText = 'Congratulations! Your account has been credited with $0.50.';

            document.getElementById('res-msg').innerText = reason || 'The quiz was not completed. Keep practicing to earn!';
            reward.style.display = 'none';
        }
        updateUI();
    }
